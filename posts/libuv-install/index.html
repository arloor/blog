иҜҙжҳҺ

иҝҷдёӘдҫӢеӯҗеҲӣе»әдәҶдёҖдёӘloopпјҢиҜҘloopз»ҸеҺҶдәҶmallocгҖҒinitгҖҒrunгҖҒcloseгҖҒfreeдә”дёӘйҳ¶ж®өпјҢе…¶дёӯmallocе’ҢfreeжҳҜcиҜӯиЁҖеҶ…еӯҳз®ЎзҗҶпјҢinitгҖҒrunгҖҒcloseеҲҷжҳҜloopзҡ„з”ҹе‘Ҫе‘ЁжңҹгҖӮ

иҝҷдёӘзЁӢеәҸдјҡй©¬дёҠйҖҖеҮәпјҢеӣ дёәloopдёҠжІЎжңүжіЁеҶҢзӣ‘еҗ¬зҡ„дәӢд»¶гҖӮ

еёҰжңүidle handleзҡ„loop

иҝҷдёӘдҫӢеӯҗз»ҷloopеўһеҠ дёҖдёӘidle handleпјҢеҸҜд»Ҙи®ӨдёәиҝҷжҳҜcpuиҝӣе…Ҙз©әй—Ізҡ„дәӢд»¶гҖӮ

дёӢйқўиҝҷдёӘдҫӢеӯҗдҪҝз”Ёuv_default_loop()д»ЈжӣҝдәҶдёҠдёӘдҫӢеӯҗзҡ„mallocе’ҢinitпјҢеҗҢж—¶д№ҹжІЎжңүжҳҫејҸеҫ—freeиҝҷдёӘdefault loopгҖӮ

иҝҷдёӘдҫӢеӯҗзҡ„йҮҚзӮ№жҳҜеҲӣе»әдәҶдёҖдёӘеҗҚдёәвҖңidlerвҖқзҡ„handleпјҢе№¶е°Ҷе…¶жіЁеҶҢеҲ°loopдёӯпјҢз„¶еҗҺеҗҜеҠЁloopгҖӮд»Јз ҒеҰӮдёӢпјҡ

    uv_idle_t idler;

   uv_idle_init(uv_default_loop(), &idler);
   uv_idle_start(&idler, count_check_stop);
  1. uv_idle_t idler;вҖ”вҖ”еңЁж ҲдёҠеҲҶй…ҚдёҖдёӘidle handle
  2. uv_idle_init(uv_default_loop(), &idler);вҖ”вҖ”жіЁеҶҢиҜҘhandleеҲ°default loop
  3. uv_idle_start(&idler, count_check_stop);вҖ”вҖ”жіЁеҶҢиҜҘhandleзҡ„еӣһи°ғдёәcount_check_stopеҮҪж•°
  4. uv_run(uv_default_loop(), UV_RUN_DEFAULT);вҖ”вҖ”loopејҖе§ӢеҫӘзҺҜ
  5. ……

uv_runжү§иЎҢеҗҺпјҢжҜҸеҪ“idleдәӢд»¶еҸ‘з”ҹж—¶пјҢйғҪдјҡи°ғз”Ёcount_check_stopеҮҪж•°пјҢдјҡеҜ№и®Ўж•°еҷЁеҠ дёҖпјҢеҰӮжһңи®Ўж•°еҷЁеӨ§дәҺ5000000пјҢеҲҷдјҡuv_idle_stopиҝҷдёӘhandleпјҲз§»йҷӨиҜҘhandleпјүгҖӮиҝҷеҜјиҮҙloopдёӯжІЎжңүhandleзӯүеҫ…и§ҰеҸ‘пјҢжүҖд»Ҙloopд№ҹдјҡйҖҖеҮәж•ҙдёӘиҝӣзЁӢйҖҖеҮәгҖӮ

жҖ»з»“дёҖдёӢпјҢеңЁloopзҡ„mallocгҖҒinitеҗҺпјҢйңҖиҰҒжіЁеҶҢhandleпјҢе®ҢжҲҗhandleзҡ„еҶ…еӯҳеҲҶй…ҚгҖҒinitпјҲжіЁеҶҢеҲ°loopпјүгҖҒstartпјҲжіЁеҶҢеӣһи°ғеҮҪж•°пјүгҖӮйңҖиҰҒз§»йҷӨиҜҘhandleж—¶еҜ№иҜҘhandleжү§иЎҢstopгҖӮ

#include <stdio.h>
#include <uv.h>

int64_t counter = 0;

void count_check_stop(uv_idle_t* handle) {
    counter++;

    if(counter%1000000==0){
        printf("%ld idle\n",counter/1000000);

    }

    if (counter >= 5000000) {
        printf("Idle stop!");
        uv_idle_stop(handle);
    }
}

int main() {
    uv_idle_t idler;

    uv_idle_init(uv_default_loop(), &idler);
    uv_idle_start(&idler, count_check_stop);

    printf("Idling...\n");
    uv_run(uv_default_loop(), UV_RUN_DEFAULT);

    uv_loop_close(uv_default_loop());
    return 0;
}

echoserverзҡ„е®һзҺ°

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <uv.h>

#define DEFAULT_PORT 7000
#define DEFAULT_BACKLOG 128

uv_loop_t *loop;
struct sockaddr_in addr;
const char *ENTER = "иҜ·иҫ“е…Ҙд»»ж„Ҹж–Үеӯ—пјҡ\n";
const char *echo = "ECHO: ";

typedef struct {
    uv_write_t req;
    uv_buf_t buf;
} write_req_t;

void free_write_req(uv_write_t *req) {
    write_req_t *wr = (write_req_t *) req;
    free(wr->buf.base);
    free(wr);
}

void alloc_buffer(uv_handle_t *handle, size_t suggested_size, uv_buf_t *buf) {
    buf->base = (char *) malloc(suggested_size);
    buf->len = suggested_size;
}

void echo_write(uv_write_t *req, int status) {
    if (status) {
        fprintf(stderr, "Write error %s\n", uv_strerror(status));
    }
    free_write_req(req);
}

void first_write(uv_write_t *req, int status) {
    if (status) {
        fprintf(stderr, "Write error %s\n", uv_strerror(status));
    } else {
        fprintf(stderr, "жҸҗзӨәеҸ‘йҖҒжҲҗеҠҹ\n");
    }

    write_req_t *wr = (write_req_t *) req;
    free(wr);
}

void echo_read(uv_stream_t *client, ssize_t nread, const uv_buf_t *buf) {

    if (nread > 0) {
        //жҲӘеҸ–char*пјҢеўһеҠ \0пјҢд»…д»…жҳҜдёәдәҶжү“еҚ°зҡ„ж—¶еҖҷдёҚеӨҡжү“пјҢеҰӮжһңз”ЁдәҺдёҡеҠЎдёҚиҰҒеҠ иҝҷдёӘпјҢиҝҷжҳҜж•°з»„и¶Ҡз•Ң
        buf->base[nread] = 0;
        fprintf(stderr, "%ld:%s", nread, buf->base);
        // еҲӣе»әеҶҷbufпјҢеўһеҠ ECHOжҸҗзӨә
        int length = strlen(echo) + nread;
        char *buff = (char *) calloc(length, sizeof(char));
        int i;
        for (i = 0; i < strlen(echo); i++) {
            buff[i] = echo[i];
        }
        for (int j = 0; j <nread ; ++j) {
            buff[j+i]=buf->base[j];
        }
        // йҮҠж”ҫиҜ»еҲ°зҡ„buf
        free(buf->base);
        write_req_t *req = (write_req_t *) malloc(sizeof(write_req_t));
        req->buf = uv_buf_init(buff, length);// еҶҷзҡ„buffйңҖиҰҒеңЁеҶҷеӣһи°ғдёӯжүҚиғҪfreeпјҢеҗҰеҲҷжҠҘй”ҷ
        uv_write((uv_write_t *) req, client, &req->buf, 1, echo_write);
        return;
    }
    if (nread < 0) {
        if (nread != UV_EOF)
            fprintf(stderr, "Read error %s\n", uv_err_name(nread));
        uv_close((uv_handle_t *) client, NULL);
    }

    free(buf->base);
}

void on_new_connection(uv_stream_t *server, int status) {
    if (status < 0) {
        fprintf(stderr, "New connection error %s\n", uv_strerror(status));
        // error!
        return;
    }

    uv_tcp_t *client = (uv_tcp_t *) malloc(sizeof(uv_tcp_t));
    uv_tcp_init(loop, client);
    if (uv_accept(server, (uv_stream_t *) client) == 0) {
        //еҸ‘йҖҒжҸҗзӨәдҝЎжҒҜ
        write_req_t *req = (write_req_t *) malloc(sizeof(write_req_t));
        req->buf = uv_buf_init((char *) ENTER, strlen(ENTER));
        uv_write((uv_write_t *) req, (uv_stream_t *) client, &req->buf, 1, first_write);

        uv_read_start((uv_stream_t *) client, alloc_buffer, echo_read);
    } else {
        uv_close((uv_handle_t *) client, NULL);
    }
}

int main() {
    loop = uv_default_loop();

    uv_tcp_t server;
    uv_tcp_init(loop, &server);

    uv_ip4_addr("0.0.0.0", DEFAULT_PORT, &addr);

    uv_tcp_bind(&server, (const struct sockaddr *) &addr, 0);
    fprintf(stderr, "ејҖе§Ӣзӣ‘еҗ¬%dпјҢиҜ·ж–°ејҖе‘Ҫд»ӨиЎҢиҫ“е…Ҙпјҡnc localhost %d\n", DEFAULT_PORT, DEFAULT_PORT);
    int r = uv_listen((uv_stream_t *) &server, DEFAULT_BACKLOG, on_new_connection);
    if (r) {
        fprintf(stderr, "Listen error %s\n", uv_strerror(r));
        return 1;
    }
    return uv_run(loop, UV_RUN_DEFAULT);
}
