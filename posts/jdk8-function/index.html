дёҚз”Ёе°ұдёҚдјҡдәҶи§ЈпјҢдёҚдәҶи§Је°ұдёҚдјҡз”ЁпјҢд»ҠеӨ©жқҘи®°дёҖдёӢJava8зҡ„lambdaиЎЁиҫҫејҸзӣёе…ізҡ„дёҖзӮ№зӮ№дёңиҘҝгҖӮ

е…ҲдёҠд»Јз Ғ

//Main.java
import java.util.function.Consumer;

public class Main {
    public static void main(String[] args) {
        System.out.println(dosometing(Model::aFunction));

        Model model = new Model();
        dosometing(model::aConsume);
    }

    private static String dosometing(IConvert<String, String> function) {
        return function.convert("war");
    }
    private static void dosometing(Consumer<String> function) {
        function.accept("war");
    }
}

//Model.java
public class Model {

    public void aConsume(String raw) {
        System.out.println("consume");
    }

    public static String aFunction(String raw) {
        System.out.println("function");
        return raw;
    }

    public String aProvide() {
        System.out.println("provide");
        return "provide";
    }
}

//IConvert.java
@FunctionalInterface
interface IConvert<F, T> {
    T convert(F form);
}

и§ЈйҮҠ

дёҠйқўд»Јз Ғжңүд»Җд№ҲдёңиҘҝпјҹиғҪзңӢеҲ°дёҖдёӘ@FunctionalInterfaceжіЁи§Јзҡ„жҺҘеҸЈIConvertгҖӮиҝҳиғҪзңӢеҲ°Model::aFunctionдҪңдёәIConvertзұ»еһӢзҡ„еҸӮж•°дј з»ҷдәҶdoSomethingж–№жі•гҖӮModel::aFunctionжҢҮзҡ„жҳҜModelзұ»зҡ„йқҷжҖҒж–№жі•aFunctionгҖӮmodel::aConsumeжҢҮзҡ„жҳҜmodelеҜ№иұЎзҡ„йқһйқҷжҖҒж–№жі•aConsumeгҖӮ

д№ҹе°ұжҳҜиҜҙModel::aFunctionжҳҜIConvertзұ»еһӢзҡ„гҖӮдёәд»Җд№ҲJDK8еј•е…ҘиҝҷдёӘи®ҫе®ҡпјҹ

вҖңйёӯеӯҗжЁЎеһӢвҖқвҖ”вҖ”иө°еҫ—еғҸйёӯеӯҗпјҢйӮЈд»–е°ұжҳҜйёӯеӯҗгҖӮиҝҷжҳҜgoиҜӯиЁҖйҮҢдҪҝз”Ёзҡ„InterfaceеҺҹзҗҶвҖ”вҖ”иЎҢдёәз¬ҰеҗҲд»Җд№Ҳзұ»еһӢпјҢе°ұжҳҜд»Җд№Ҳзұ»еһӢгҖӮеңЁjavaйҮҢinterfaceжҺҘеҸЈе®ҡд№үзҡ„е°ұжҳҜиЎҢдёәгҖӮиҝҷйҮҢзҡ„IConvertе®ҡд№үдәҶдёҖдёӘиЎҢдёәconvertпјҢд»ҘFдёәеҸӮж•°пјҢд»ҘTдёәиҫ“еҮәгҖӮиҖҢModel::aFunctionиҝҷдёӘйқҷжҖҒж–№жі•жүҖе®ҡд№үзҡ„жӯЈжҳҜд»ҘFдёәеҸӮж•°пјҢд»ҘTдёәиҫ“еҮәпјҲжӯЈзЎ®и®ҫзҪ®жіӣеһӢзҡ„еүҚжҸҗдёӢпјүпјҢд№ҹе°ұжҳҜиҜҙModel::aFunctionиҝҷдёӘйқҷжҖҒж–№жі•з¬ҰеҗҲconvertиЎҢдёәпјҢеңЁйёӯеӯҗжЁЎеһӢдёӢпјҢModel::aFunctionе°ұжҳҜIConvertгҖӮ

еҪ“з„¶IConvertжҺҘеҸЈд№ҹжҳҜжңүдёҖдәӣйҷҗеҲ¶зҡ„пјҢд»–йңҖиҰҒжҳҜдёҖдёӘFunctionalInterfaceвҖ”вҖ”еҸӘжңүдёҖдёӘжҠҪиұЎж–№жі•зҡ„жҺҘеҸЈгҖӮд№ҹе°ұжҳҜд»–еҸӘиғҪжңүдёҖдёӘжңӘе®ҡд№үж–№жі•гҖӮеҪ“з„¶д»–д№ҹжҳҜеҸҜд»Ҙжңүdefaultе’Ңstaticж–№жі•зҡ„гҖӮжүҖд»ҘIConvertжҺҘеҸЈд№ҹеҸҜд»ҘеҰӮдёӢпјҡ

@FunctionalInterface
interface IConvert<F, T> {
    T convert(F form);

    static void some() {
        System.out.println("staticж–№жі•");
    }
    default void another() {
        System.out.println("defaultж–№жі•");
    }
}

дёӨеҸҘиҜқжҖ»з»“пјҡ

  1. Class::methodеҸҜд»ҘдҪңдёәжҹҗInterfaceзҡ„е®һдҫӢпјҢд»ҺиҖҢз”ЁдҪңеҸӮж•°жҲ–иҖ…иҝ”еӣһеҖј
  2. иҜҘInterfaceйңҖиҰҒз¬ҰеҗҲ@FunctionalInterfaceпјҢд№ҹе°ұжҳҜеҸӘжңүдёҖдёӘжҠҪиұЎж–№жі•

java.util.functionеҢ…

еғҸIConvertдёҖж ·зҡ„@FunctionalInterfaceжҺҘеҸЈе…¶е®һжҳҜжңүдёҖдәӣеҫҲйҖҡз”Ёзҡ„пјҢjdkеҶ…зҪ®дәҶиҝҷдәӣжҺҘеҸЈпјҢе°ұж”ҫеңЁjava.util.functionеҢ…дёӯгҖӮ

жңүFunctionгҖҒConsumerгҖҒProviderиҝҷдәӣйҖҡз”Ёзҡ„зұ»

еҖјеҫ—жіЁж„Ҹзҡ„з”Ёжі•пјҡд»Ҙ@FunctionalInterfaceдёәиҝ”еӣһеҖј

д»ҘвҖңеҠҹиғҪжҺҘеҸЈвҖқдҪңдёәеҸӮж•°зҡ„з”Ёжі•еҫҲеёёз”ЁпјҢжҜ”еҰӮеҗ„з§Қеӣһи°ғеҮҪж•°пјҢEventListenrзӯүзӯүгҖӮ

д»ҘвҖңеҠҹиғҪжҺҘеҸЈвҖқдҪңдёәиҝ”еӣһеҖјпјҲд№ҹе°ұжҳҜвҖңеҠҹиғҪжҺҘеҸЈвҖқе·ҘеҺӮж–№жі•пјүпјҢеҲҷеҸҜд»ҘжӣҙеҠ иҮӘз”ұең°з»ҷвҖңеҠҹиғҪжҺҘеҸЈвҖқеЎһе…ҘдёҖдәӣеұһжҖ§пјҢе®ҢжҲҗе®ҡеҲ¶еҢ–гҖӮ

    public static <T, E extends Enum<E>> Function<T, E> lookupMap(E[] values, Function<E, T> mapper) {
        Map<T, E> index = Maps.newHashMapWithExpectedSize(values.length);
        for (E value : values) {
            index.put(mapper.apply(value), value);
        }
        return (T key) -> index.get(key);
    }

иҝҷж ·дёҖж®өд»Јз ҒпјҢеҲӣе»әдәҶдёҖдёӘеҢ…еҗ«Mapзҡ„FuncitonгҖӮиҝҷе…¶е®һжҳҜдёҖдёӘй—ӯеҢ…пјҢиҝ”еӣһзҡ„FunctionеҜ№иұЎеҸҜд»ҘдҪҝз”ЁindexиҝҷдёӘеј•з”ЁгҖӮеӣ дёәеҶ…йғЁзұ»еҸҜд»ҘдҪҝз”ЁеӨ–йғЁеҜ№иұЎзҡ„еҸҳйҮҸгҖӮвҖ”вҖ”javaзҡ„еҶ…йғЁзұ»е°ұжҳҜй—ӯеҢ…е®һзҺ°гҖӮ

д»Җд№ҲжҳҜй—ӯеҢ…пјҹвҖ”вҖ”й—ӯеҢ…еҸҜд»Ҙи®©дҪ д»ҺеҶ…йғЁеҮҪж•°и®ҝй—®еӨ–йғЁеҮҪж•°дҪңз”ЁеҹҹгҖӮз”ҹеҠЁеҪўиұЎдё”иҜҰз»Ҷе№¶дёҫдҫӢеӯҗең°и§ЈйҮҠй—ӯеҢ…

жҲ‘д»¬е…ҲзңӢдёҖж®өjsд»Јз Ғпјҡ

function makeFunc() {
    var name = "Mozilla";
    function displayName() {
        alert(name);
    }
    return displayName;
}

var myFunc = makeFunc();
myFunc();

еј•з”Ёпјҡз¬¬дёҖзңјзңӢдёҠеҺ»пјҢд№ҹи®ёдёҚиғҪзӣҙи§Ӯзҡ„зңӢеҮәиҝҷж®өд»Јз ҒиғҪеӨҹжӯЈеёёиҝҗиЎҢгҖӮеңЁдёҖдәӣзј–зЁӢиҜӯиЁҖдёӯпјҢеҮҪж•°дёӯзҡ„еұҖйғЁеҸҳйҮҸд»…еңЁеҮҪж•°зҡ„жү§иЎҢжңҹй—ҙеҸҜз”ЁгҖӮдёҖж—Ұ makeFunc() жү§иЎҢе®ҢжҜ•пјҢжҲ‘д»¬дјҡи®Өдёә name еҸҳйҮҸе°ҶдёҚиғҪиў«и®ҝй—®гҖӮз„¶иҖҢпјҢеӣ дёәд»Јз ҒиҝҗиЎҢеҫ—жІЎй—®йўҳпјҢжүҖд»ҘеҫҲжҳҫз„¶еңЁ JavaScript дёӯе№¶дёҚжҳҜиҝҷж ·зҡ„гҖӮ

жҲ‘д»¬зҡ„javaе°ұжҳҜиҝҷз§ҚеұҖйғЁж ҮйҮҸд»…еңЁеҮҪж•°жү§иЎҢжңҹжүҚеҸҜз”Ёзҡ„иҜӯиЁҖпјҢдёәд»Җд№ҲиҝҷйҮҢеҸҲеҸҜиЎҢе‘ўпјҹпјҹпјҹпјҹпјҹпјҹпјҹпјҹпјҹеӣ дёәjavaеҶ…йғЁзұ»еҸҜд»Ҙи®ҝй—®еӨ–йғЁзұ»зҡ„еҸҳйҮҸгҖӮ

е°Ҹи®°

д»ҠеӨ©жҳҜ1024зЁӢеәҸе‘ҳиҠӮпјҢдёҠеӯҰзҡ„ж—¶еҖҷе°ұзҹҘйҒ“иҝҷдёӘиҠӮпјҢдҪҶд»ҺжқҘжІЎеңЁж„ҸиҝҮгҖӮе·ҘдҪңд»ҘжқҘзҡ„з¬¬дёҖдёӘзЁӢеәҸе‘ҳиҠӮж„ҹи§үиҝҳжҳҜжҢәеҘҪзҡ„гҖӮ

  • з¬¬дёҖеј№пјҡзЁӢеәҸе‘ҳзӨјзӣ’{е°ҸжӢӣе–өе®ҡеҲ¶еҚЎеҘ—гҖҒCodingдё»йўҳиӮҘе®…еҝ«д№җж°ҙ}
  • з¬¬дәҢеј№пјҡжҳҹе·ҙе…Ӣй…ёеҘ¶+иӣӢзі•
  • з¬¬дёүеј№пјҡе…Қиҙ№жҷҡйӨҗ{йёӯиЎҖзІүдёқжұӨгҖҒйҰ„йҘЁгҖҒе°ҸиӣӢзі•гҖҒйёЎи…ҝ}

еӣӣдёӘејҖеҸ‘е®Өзҡ„иЎЁжј”пјҢж‘Үж»ҡжҳҜзңҹзҡ„highпјҢзӣёеЈ°д№ҹиҝҳжҳҜиғҪйЎ¶зҡ„дҪҸзҡ„е•ҠгҖӮгҖӮгҖӮ

дёҠдёӘе‘Ёжң«еҲҡеҘҪжҳҜз§‘е®Өз§ӢжёёпјҢеңЁеӨ–йқўдҪҸдәҶдёҖжҷҡпјҢеҗғдәҶдёүйЎҝйҘӯпјҢеҒҡдәҶдёҖдәӣжҙ»еҠЁпјҢдёҖеҸҘиҜқж°ӣеӣҙзңҹзҡ„еҘҪгҖӮ
